x86:

1. Запускаем qt5vars.cmd
2. cd qtbase
   configure -opensource -opengl desktop -mp -nomake tests -nomake examples -target xp
   nmake

3. cd ..\qttools
   qmake
   nmake

4.  cd ..\qtbase
    nmake docs
    cd ..\qttools
    nmake docs
 
5. cd ..
   cleanup.bat
